2. Le composant

function Dashboard() {
 // 🧠 STATE — la mémoire
 const [filtre, setFiltre] = useState("")

 // ⚙️ CALCUL — comme df.query()
 const lignes = donnees.filter(d =>
  d.region.includes(filtre)
 )

 // 🖥️ RENDU — l'interface
 return (
  <input onChange={setFiltre} />
  <Tableau data={lignes} />
 )
}
